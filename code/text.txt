#Calcular factorial
def calcularFac(num):
    num_aux = 0
    if (num < 1):
        num_aux = -1 #No se contemplaron los numeros negativos en los tokens de numeros
              # Token anterior: FLOAT : NUM'.'NUM;  Nuevo Token: INTEGER : '-'?NUM;  FLOAT : '-'?NUM'.'NUM
    else:
        num_aux = num * (calcularFac(num--1)) #Numero negativo no contemplado, los token de numeros ya fueron arreglados
    return num_aux
ventas = [100,200,300,400,500,-3.14] #Numero negativo  no contemplado, los token de numeros ya fueron arreglados

Caracter = '%'

def promedio(cualquier_arreglo):
    tam = len(cualquier_arreglo)
    resultado=0
    #ciclo para recorrer arreglo
    cont = 0
    sumatoria = 0
    while (cont <= tam-1):
        sumatoria = sumatoria + cualquier_arreglo[cont]
        cont = cont + 1
    if tam > 0:
        resultado = sumatoria / tam
    return resultado

def calcularpromedioventas():
    prom = promedio(len(ventas))
    print("El promedio de las ventas es: ")
    print('\n')
    print(prom)
